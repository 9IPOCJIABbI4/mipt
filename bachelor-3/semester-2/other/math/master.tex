\begin{gather*}

\int \frac{dx}{\sqrt{x^2+a^2} }=\ln \left| x+\sqrt{x^2+a^2}  \right| +C

Дифференциальный бином:
\[
	I=\int x^m\left( a+b x^n \right) ^p dx,\ \left(m,\,n,\,p \in \mathbb{Q}\right)
.\] 
Замены:
\begin{itemize}
	\item $p \in \mathbb{Z}$:\\
	 $x=t^k$,  $k$ ---
		общ. знам.  $m$ и $n$
	\item $\frac{m+1}{n} \in \mathbb{Z}$:\\
	 $a+bx^n=t^s$,
		 $s$ --- знам. $p$
	 \item $p+ \frac{m+1}{n} \in \mathbb{Z}$:\\
	 $a x^{-n}+b=t^s$, $s$ --- знам. $p$.
\end{itemize}
